function getRecruitment(isRight, targetYaw, targetPitch) { const side = isRight ? 'right' : 'left'; const prefix = isRight ? 'R-' : 'L-'; // Health multipliers for each nerve/muscle unit [cite: 34] const h = { LR: SYSTEM_STATE.nerves[prefix+'CN6'] * SYSTEM_STATE.muscles[side].LR, MR: SYSTEM_STATE.nerves[prefix+'CN3'] * SYSTEM_STATE.muscles[side].MR, SR: SYSTEM_STATE.nerves[prefix+'CN3'] * SYSTEM_STATE.muscles[side].SR, IR: SYSTEM_STATE.nerves[prefix+'CN3'] * SYSTEM_STATE.muscles[side].IR, IO: SYSTEM_STATE.nerves[prefix+'CN3'] * SYSTEM_STATE.muscles[side].IO, SO: SYSTEM_STATE.nerves[prefix+'CN4'] * SYSTEM_STATE.muscles[side].SO }; // 1. BASAL TONE & DRIFT (The "Down and Out" Fix) [cite: 35, 36] // Horizontal: Unopposed LR pulls OUT[cite: 35]. const driftX = (1 - h.LR) * -0.4 + (1 - h.MR) * 0.4; // Vertical: Unopposed SO pulls DOWN[cite: 4, 36, 45]. // If CN3 (SR and IR) is out, drift is significantly Negative (Down) due to SO tone[cite: 4, 36]. const driftY = (1 - h.SR) * -0.1 + (1 - h.IR) * 0.1 + (h.SR === 0 && h.IR === 0 ? -0.25 : 0); // 2. SMOOTH MECHANICAL BLENDING [cite: 37, 38, 39] let allowedYaw = isRight ? (targetYaw < 0 ? targetYaw * h.LR : targetYaw * h.MR) : (targetYaw > 0 ? targetYaw * h.LR : targetYaw * h.MR); // Nasal factor smoothly scales SO depression vs IR depression [cite: 39, 40] const nasalYaw = isRight ? targetYaw : -targetYaw; const nasalFactor = THREE.MathUtils.clamp((nasalYaw + 0.5) / 1.0, 0, 1); // 3. VERTICAL AUTHORITY [cite: 40, 41, 42] let allowedPitch; if (targetPitch > 0) { allowedPitch = nasalFactor > 0.5 ? targetPitch * h.IO : targetPitch * h.SR; } else { allowedPitch = nasalFactor > 0.5 ? targetPitch * h.SO : targetPitch * h.IR; } const finalYaw = allowedYaw + (isRight ? -driftX : driftX); [cite: 42] const finalPitch = allowedPitch + driftY; [cite: 43] const abd = isRight ? -finalYaw : finalYaw; [cite: 43] const add = -abd; [cite: 43] // 4. ACTS / MUSCLE BAR PERCENTAGES [cite: 44, 45, 52] // In CN III palsy, SO acts as a primary depressor in this state[cite: 4, 45]. return { rotation: { y: finalYaw, x: finalPitch }, acts: { LR: (0.2 + Math.max(0, abd) * 1.8) * h.LR, MR: (0.2 + Math.max(0, add) * 1.8) * h.MR, SR: (0.2 + Math.max(0, finalPitch) * 2.2 * (1 - nasalFactor)) * h.SR, IR: (0.2 + Math.max(0, -finalPitch) * 1.8 * (1 - nasalFactor)) * h.IR, IO: (0.2 + Math.max(0, finalPitch) * 2.0 * nasalFactor) * h.IO, SO: (0.2 + Math.max(0, -finalPitch) * 1.8 * nasalFactor + (h.IR === 0 ? 0.3 : 0)) * h.SO } }; }
